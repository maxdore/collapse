∂ n = ℕSC (ssubsets (n -to-0)) closed where
  closed : ssubsets (n -to-0) closedUnder _⊆_
  closed X P Y Q = ssubsets-comp (⊆⊂-trans Y X (n -to-0) Q (ssubsets-corr P))
